			wait(30)

			MES(ANNA,"何よあんたたち。\n見ない顔ねー。",0)

		}

		//　　※以下通常一般メッセージ。

		{

					MES_close(RAGNA)
				}
				else

					MES_close(PARTNER)
				}
				//──────────────────────────────

			}
			//　　★繰り返し
			else
			{
				MES(ANNA,"あの子たちが来てから、\nなかなかサボる隙が見つからないの。",0)
				MES(ANNA,"あーあ、やってらんないわ～。",0)
				MES_close(ANNA)
			}
			rot(ANNA,30,90)
		}
		MOT(ANNA,2,15)
	}
	
	TK_end()
}
