	color(PET,-1)//透過

	EV("Lue_Set")

	//フェードイン----------------------
	fade_in(0,60,0)
	wait_fade()
	//――――――――――――――――――――――――――
	//	イベント開始
	//――――――――――――――――――――――――――

	MOT(ALWEN,133,15)

	new_map(10440)

	//プレイヤーは消す
	//アルウェンがいる
	if(F6_00_GoToMeru)
	{
		chr_pos(RAGNA,-839,790,-48,0,2)
		chr_pos(PARTNER,-735,678,-50,0,2)
		chr_pos(PET,-575,627,-50,0,2)
		color(RAGNA,-1)//透過
		color(PARTNER,-1)//透過
		color(PET,-1)//透過
	}
	else
	//オデッサがいる
	if(F5_08_JoinOdessa && !F5_13_KillDygald)
	{
		chr_pos(RAGNA,-839,790,-48,0,2)
		chr_pos(PARTNER,-735,678,-50,0,2)
		chr_pos(PET,-575,627,-50,0,2)
		color(RAGNA,-1)//透過
		color(PARTNER,-1)//透過
		color(PET,-1)//透過
	}
	else
	{
		chr_pos(RAGNA,-839,790,-48,0,2)
		chr_pos(PET,-575,627,-50,0,2)
		color(RAGNA,-1)//透過
		color(PET,-1)//透過
	}
	
	//モーション
	MOT_SET(EVENT_RAGNA,150,-2,521,521,-1,-1)		//座り

	MOT(EVENT_RAGNA, 150, 0 )

	//フェードイン----------------------
	fade_in(0,60,0)
	wait_fade()
	//――――――――――――――――――――――――――
	//	イベント開始
	//――――――――――――――――――――――――――
	KAO(EVENT_RAGNA,"1","I","B")
	look(EVENT_RAGNA,"Bone_Head",0,		0,2,2,			0,0,45)
	wait(50)
	KAO(EVENT_RAGNA,"1","J","1")
	wait(60)

	workL(0,RAND,5)
	if(WK000==0)
	{
		//アルウェンがいる
		if(F6_00_GoToMeru)
		{
			EV("Alwen_Set")
			wait_EV("Alwen_Set")
		}
		else
		//オデッサがいる
		if(F5_08_JoinOdessa && !F5_13_KillDygald)
		{
			EV("Odessa_Set")
			wait_EV("Odessa_Set")
		}
	}

	wait(120)
	
	//――――――――――――――――――――――――――
	//	イベント終了
	//――――――――――――――――――――――――――
	fade_in(100,30,0)
	wait_fade()
	
	//脱衣所マップへ
	workG(GW_ENTRY_EVENT,SET,1)
	new_map(10440)
}
