	// 登場

	//     set deg tim rnd  len next
	B_NEXT  80   0  60   0    0   02	//⇒始動


	//------------------------------------------------------------------
	// 待機
	//------------------------------------------------------------------
	//    set typ spd rot xx1 xx2
	B_MOV  02   0   0   4  00  00
	B_MOV  13   0   0   4  00  00

	if_SET 13
	{
//		if_HP 30 0
//		{
//			if_mFx= 0
//			{
//				mFx_set -1 -1 1
//
//				//     set deg tim rnd  len next
//				B_NEXT  -1   0   0   0    0   40		//⇒逃走
//			}
//		}

		if_HP 100 0
		{
			if_mF0= 0
			{
				mF0_add -1 -1 1
				
				//     set deg tim rnd  len next
				B_NEXT  -1   0   0   0    0   33		//⇒炎の雨
			}
			el_mF0= 1
			{
				mF0_add -1 -1 1
				
				//     set deg tim rnd  len next
				B_NEXT  -1   0   0   0    0   32		//⇒Ｖ字炎
			}
			el_mF0= 2
			{
				mF0_add -1 -1 1
				
				//     set deg tim rnd  len next
				B_NEXT  -1   0   0   0    0   35		//⇒Ｖ字炎２
			}
			el_mF0< 6
			{
				mF0_add -1 -1 1
				
				//     set deg tim rnd  len next
				B_NEXT  -1   0   0   0    0   01		//⇒体当り
			}
			el_mF0= 6
			{
				mF0_set -1 -1 0
			}
		}
	}


	//------------------------------------------------------------------
	// 攻撃（炎の雨）

	//    set typ spd rot xx1 xx2
	B_MOV  33   0   0   8  00  00
	B_MOV  34   0   0   8  00  00
	


	//------------------------------------------------------------------

	//------------------------------------------------------------------
	// 攻撃（体当り）
	//------------------------------------------------------------------
	//    set typ spd rot xx1 xx2
	B_MOV  01   0   0  20  00  00			//高速で旋回
	B_MOV  11   0  30  12  00  00			//高速で突撃（旋回不可）
		//     set deg tim rnd len next
		B_NEXT  11   0  60   0   0   12		//一定時間後、停止

	if_SET 11
	{
		if_EV TOUCH_MAP
		{
			//     set deg tim rnd len next
			B_NEXT  -1   0   0   0   0   12		//⇒停止
		}

		if_EV HIT
		{
			E_ADL -1 -1 1180 185 -1 -1		//＠ぶちかまし衝撃

			//     set deg tim rnd len next
			B_NEXT  -1   0   0   0   0   12		//⇒停止
		}

		if_time 8
		{
			OTO -1 -1 441 0 0 0		//突撃ＳＥ
			OTO -1 -1 421 0 0 0		//突撃ＳＥ
		}
	}

		}

		}
