#ScriptTable

TK_DAIGORO
{
	TK_begin()
////////////////////////////////////////////////////////////////
//仲間全員が話し相手に振り向く
////////////////////////////////////////////////////////////////
	KAO(DAIGORO, "1", "1", "1")	
	KAO(PLAYER1, "1", "1", "1")	
	rot_chr(PLAYER1,30,DAIGORO)
//───────────────
//⑪ルゥが居る
//───────────────
	if(F5_00_HearLunar)
	{
		KAO(PET, "1", "1", "1")	
		rot_chr(PET,30,DAIGORO)
		
		//オデッサがいる
		if(F5_08_JoinOdessa && !F5_13_KillDygald)
		{
			KAO(PLAYER2, "1", "1", "1")	
			rot_chr(PLAYER2,30,DAIGORO)
		}
		//アルウェンがいる
		else
		if(F6_00_GoToMeru)
		{
			KAO(PLAYER2, "1", "1", "1")	
			rot_chr(PLAYER2,30,DAIGORO)
		}
	}
//───────────────
//⑦一人で行動
//───────────────
	else
	if(F4_00_StartSolo)
	{
	//───────────────
	//スバルが居る
	//───────────────
		if(F4_02_GoVillage2 && !F4_06_StopAttack)
		{
			KAO(PLAYER2, "1", "1", "1")	
			rot_chr(PLAYER2,30,DAIGORO)
		}
		else
	//───────────────
	//アルウェンが居る
	//───────────────
		if(F4_09_GoShrineMia)
		{
			KAO(PLAYER2, "1", "1", "1")	
			rot_chr(PLAYER2,30,DAIGORO)
		}
	}
//会話処理へ
////////////////////////////////////////////////////////////////
	
	//──────────────────────────────
	//▼⑬
	//──────────────────────────────
	if(F6_05_EndSixth)
	{
//　　★ダイゴロー【宿酒場に遊びに来た】
	//　　※アルッテの宿酒場、ルリの傍に立っている。
	//　　※振り返らない。
		MES(DAIGORO,"あうー……",0)
		MES(DAIGORO,"これ、これたべりゅ……！",0)
		MES(DAIGORO,"あうあうっ！",0)
		MES_close(DAIGORO)
		//★ダイゴロー【宿酒場に遊びに来た】
		F_set_note(DAIGORO,2)
	}
	else
	//──────────────────────────────
	//▼⑫飛行機でメルセデクに乗り込むことにした（ラグナ・アルウェン・ルゥ）
	//──────────────────────────────
	//　　※里の階段上辺りでフリームーブ。
	if(F6_00_GoToMeru)
	{
		MOT(DAIGORO,2,10)
		MOT_SET(DAIGORO,150,-1,478,478,151,25)	//ダイゴロー
		MOT_SET(DAIGORO,151,-1,476,476,-1,10)

		rot_chr(DAIGORO,30,PLAYER1)
		MES(DAIGORO,"あうあう。",0)
		MES(DAIGORO,"あうう～……",0)
		MES_close(DAIGORO)

	//　　ずずーっというＳＥと共に、鼻をすする。
		MOT(DAIGORO,150,15)
		wait(5)
	//　　鼻はすぐに垂れてくる。
//		MES(DAIGORO,"C4◆ずずーっというＳＥと共に、鼻をすする。\n鼻はすぐに垂れてくる。",0)
		MES(DAIGORO,"ずずーっ……",0)
		MES_close(DAIGORO)
		MOT(DAIGORO,2,10)
		wait(20)
	
		MES(DAIGORO,"あぶぅ……",0)
		MES_close(DAIGORO)
	}
	else
	//──────────────────────────── ──
	//▼⑪クリスタルバレーを起動することにした（ラグナ・ルゥ（・オデッサ））
	//──────────────────────────────
	if( F5_00_HearLunar )
	{
	//　　※出店跡で、カラスに懐いている。（手を引っ張るなど）
	//　　※振り返らない。
		MES(DAIGORO,"あーあ……",0)
		MES(DAIGORO,"あだだぁー！",0)
		MES_close(DAIGORO)
	}
	else
	//──────────────────────────────
	//▼⑩社へ向かったミアを追うことにした（ラグナ・アルウェン）
	//──────────────────────────────
	//　　※いない。
	//──────────────────────────────
	//▼⑨汎用忍者が倒れているだけのフラグ。
	//──────────────────────────────
	//　　※いない。
	//──────────────────────────────
	//▼夕方
	//──────────────────────────────
	if( F4_17_InTheEvening )
	{
		rot_chr(DAIGORO,30,PLAYER1)
		
		MES(DAIGORO,"あー……",0)
		MES(DAIGORO,"あうあうあう～……",0)
		MES_close(DAIGORO)	
	}
	else
	//──────────────────────────────
	//▼⑧星降りの里に到着した（ラグナ・スバル）
	//──────────────────────────────
	//　　※階段下の辺りをうろうろしている。
	if( F4_05_OnVillage2 )
	{
		rot_chr(DAIGORO,30,PLAYER1)
		
		MES(DAIGORO,"あー……",0)
		MES(DAIGORO,"あうあうあう～？\n（ふもとに行く？）",0)
		MES_close(DAIGORO)
		
		if(!FS_08_TalkDaigoro)
		{
			//　　※ＧＦ『フラグ⑧ダイゴローと話した』を立てる。
			F_set(FS_08_TalkDaigoro)

		//名前付きキャラ全員に話しかけると露天風呂が開く
			EV("EV_F4_12_OkRoten")
		}
		MOT(DAIGORO,2,10)

		//メニュー表示---------------------------------------------------------
		
//		MES(DAIGORO,"未入力項目：仮メッセージ",0)
//		MES_close(DAIGORO)		
		
		workL(WK_CANCEL,SET,2)//【やめる】
		MES_pos(PLAYER1,"??W9【麓に行く】\n【ここに留まる】",0,320,240,0,0)
		wait_W(WK_YESNO)
		MES_close(PLAYER1)
		wait_MES(PLAYER1)
		
		if(WK_YESNO==1)
		{
		//	name("⇒はい")
//			MES(PLAYER1,"よし、ふもとにもどるぞ！",0)
//			MES_close(PLAYER1)
			MES(DAIGORO,"あう～！",0)
			MES_close(DAIGORO)

			fade_in(100,30,0)
			wait_fade()
						
			//麓手前マップに戻る
			workG(GW_ENTRY_EVENT,SET,3)
			new_map(16010)
			
		}
		else
		{
		//	name("⇒いいえ")
//			MES(PLAYER1,"いや……\nま、今はやめとこうぜ。",0)
//			MES_close(PLAYER1)
			MES(DAIGORO,"えう～……",0)
			MES_close(DAIGORO)
		}
		
	}
	TK_end()
}
