EV_GetKinka
{
//イベント開始
	
//　　通常通り、アイテム入手メッセージを表示。
	TK_begin()
	rot_chr(PLAYER,60,49)

//	MES(this,"『幸運の金貨』を入手した。",0)
	MOT(49,1,0)
	wait(30)
	fade_in(30,10,BLACK)

	get_item(139,1,0)
	wait(10)
	wait_key(0)
	
	fade_in(0,10,BLACK)
	menu(1,		-1,	-1,-1,-1,	-1,-1,	10,0,0)
	menu(0,		-1,	-1,-1,-1,	-1,-1,	10,0,0)

	TK_end()


//　　クロスフェードでラグナ・アルウェンを宝箱の前に再配置。
	cross_fade(30)
	EV_begin()
	

	//カメラセット
	CAM_move(7349,-9338,-19,20,0)
	CAM(390,100,-19,20,225,0,0)	
	
	//キャラセット
	chr_pos(RAGNA,7518,9350,0,60,2)
	chr_pos(PARTNER,7518,9623,0,,120,2)
	

	MES(RAGNA,"へえ、こいつは……\n古代の金貨みたいだな。",0)
	MES(RAGNA,"そういや古い金貨を持ってると\n幸運がやってくるって言うが……",0)
	MES_close(RAGNA)

//	name2(PARTNER)
	rot_chr(PARTNER,60,RAGNA)
	MES(PARTNER,"ふむ、あながち\n嘘ではないかもしれんぞ。",0)
	MES_close(PARTNER)

//　　ラグナに？エモーション。
	EMO(RAGNA,EMO_HATENA)
	wait(30)

	rot_chr(RAGNA,60,PARTNER)
	
	MES(PARTNER,"人の子の言い伝えなどは\n知らぬが……",0)
	MES(PARTNER,"この金貨は、それなりの\n魔力を秘めておるようだ。",0)
	MES(PARTNER,"見れば聖刻まで入っておるし、\nこれだけの品であれば、持ち主に\n幸運をもたらす程度の事はできよう。",0)
	MES_close(PARTNER)

//	name2(RAGNA)
	MES(RAGNA,"へ、へえ、成る程な。",0)
	MES(RAGNA,"ちょっとした\n魔法の品ってワケか。",0)
	MES_close(RAGNA)

	//──────────────────────────────
	//	name("⇒クレース・フラグ③を見ている場合。")
	if(FE_03_TalkCleese)
	{
	//　　ラグナにＰＡエモーション。
		EMO(RAGNA,EMO_PA)
		wait(30)
	//　　ラグナ、宝箱の方に向きを戻す。
		rot_chr(RAGNA,60,49)
		
		MES(RAGNA,"ん？　そういや……",0)
		MES(RAGNA,"誰かがお守りの金貨を\n無くしたとか言ってたよな……",0)
		MES(RAGNA,"うーん、誰だったかな。",0)
		MES_close(RAGNA)
	}
	//	name("⇒見ていない場合")
	else
	{
	//　　ラグナ、宝箱の方に向きを戻す。
		rot_chr(RAGNA,60,49)
		MES(RAGNA,"（あれ、でもこいつ……\n  結構丁寧に手入れしてあるな。）",0)
		MES(RAGNA,"（つい最近まで使ってたみたいだし……\n  誰かの落し物か何かか……？）",0)
		MES_close(RAGNA)
	}
	//──────────────────────────────
	
//　　クロスフェードで元の座標から操作許可を戻す。
	cross_fade(30)
	//カメラを初期設定に
	CAM_return( 30 )
	wait(30)
	EV_end()
}
